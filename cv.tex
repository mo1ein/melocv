% ===== Profile photo =====
\newcommand{\profilecircle}[5][2.55cm]{%
  % #1 = circle size
  % #2 = image
  % #3 = zoom factor (1.0, 1.2, 1.5...)
  % #4 = x shift
  % #5 = y shift
  \pgfmathsetlengthmacro{\picsize}{#3*#1}
  \pgfmathsetlengthmacro{\circr}{0.5*#1-2.2pt}

  \begin{tikzpicture}
    \clip (0,0) circle[radius=\circr];
    \node[inner sep=0pt, anchor=center, xshift=#4, yshift=#5] at (0,0) {%
      \includegraphics[width=\picsize,keepaspectratio]{#2}%
    };

    \draw[
      circle,
      draw=sectioncolor!90!black,
      line width=1.15pt
    ] (0,0) circle[radius=\circr];
  \end{tikzpicture}%
}

\newcommand{\jobentry}[5][]{
  \subsection*{
    \if\relax\detokenize{#1}\relax
      {\companyfont #2}% no link
    \else
        {\companyfont \href{#1}{\texorpdfstring{#2 \tiny\faIcon{external-link-alt}}{#2}}} % linked company name
    \fi
    \hfill {\jobdatefont #3}
  }
  {\jobrolefont \textit{#4}}\hfill {\joblocfont #5} \vspace{2pt}
}

}
